 18 /**
 19 	@class CalCoreSubMorphTarget
 20 */
 21 Cal3D.CalCoreSubMorphTarget = function() {
 22 	this.m_vectorBlendVertex = [];
 23 };
 24 
 25 /**
 26 	Get the number of blend vertices.<br />
 27 	This function returns the number of blend vertices in the core sub morph target instance.
 28 	@returns {number} The number of blend vertices.
 29 */
 30 Cal3D.CalCoreSubMorphTarget.prototype.getBlendVertexCount = function() {
 31 	return this.m_vectorBlendVertex.length;
 32 };
 33 
 34 /**
 35 	Get the blend vertex list.<br />
 36 	This function returns the list that contains all blend vertices of the core sub morph target instance.
 37 	@returns {Array} The blend vertex list.
 38 */
 39 Cal3D.CalCoreSubMorphTarget.prototype.getVectorBlendVertex = function() {
 40 	return this.m_vectorBlendVertex;
 41 };
 42 
 43 /**
 44 	Allocate space for the blend vertices.<br />
 45 	This function reserves space for the blend vertices of the core sub morph target instance.
 46 	@param {number} blendVertexCount The number of blend vertices that this core sub morph target should holds.
 47 	@returns {boolean} true if succeeded; false if any error happened. 
 48 */
 49 Cal3D.CalCoreSubMorphTarget.prototype.reserve = function(blendVertexCount) {
 50 	this.m_vectorBlendVertex = new Array(blendVertexCount);
 51 	return true;
 52 };
 53 
 54 /**
 55 	Set a blend vertex.<br />
 56 	This function sets a specified blend vertex in the core sub morph target instance.
 57 	@param {number} blendVertexId The ID of the vertex.
 58 	@param {Cal3D.CalCoreSubMorphTarget.BlendVertex} blendVertex The vertex to set.
 59 	@returns {boolean} true if succeeded; false if the vertex does not exist.
 60 */
 61 Cal3D.CalCoreSubMorphTarget.prototype.setBlendVertex = function(blendVertexId, blendVertex) {
 62 	if(blendVertexId < 0 || blendVertexId >= this.m_vectorBlendVertex.length)
 63 		return false;
 64 
 65 	this.m_vectorBlendVertex[blendVertexId] = blendVertex;
 66 	return true;
 67 };
 68 
 69 
 70 
 71 /**
 72 	@class BlendVertex
 73 
 74 	The blend vertex structure for the CalCoreSubMorphTarget class.
 75 */
 76 Cal3D.CalCoreSubMorphTarget.BlendVertex = function() {
 77 	this.position = new Cal3D.CalVector;
 78 	this.normal = new Cal3D.CalVector;
 79 };
